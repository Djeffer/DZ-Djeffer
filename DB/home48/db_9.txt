1.	Выведите список студентов физико-математического факультета.
SELECT FIO
FROM Student
WHERE "Group" IS
    (SELECT "Group"
    FROM G
    WHERE Kafedra IS
        (SELECT Kafedra
        FROM Kafedra
        WHERE Decanat LIKE 'Физико-математический'
        )
    )

2.	Студенты какой кафедры и факультета сдавали социологию.
SELECT Kafedra, Decanat
FROM Kafedra
WHERE Kafedra IN
	(SELECT Kafedra
	FROM G
	WHERE "Group" IN
		(SELECT "Group"
		FROM Student
		WHERE ID_St IN
			(SELECT ID_St
			FROM Exzamen
			WHERE Predmet LIKE 'Социология')))

3.	Выведите кафедры, список студентов, список студентов в алфавитном порядке.
SELECT s.FIO, 
    (SELECT g.Kafedra
    FROM G g
    WHERE s."Group"=g."Group") AS [Kafedra]
FROM Student s
ORDER BY FIO

5.	Выведите студентов физико-технического факультета сдавших иностранный язык на 5.
SELECT FIO,
    (SELECT "Group"
    FROM G
    WHERE Kafedra IN
        (SELECT Kafedra
        FROM Kafedra
        WHERE Decanat LIKE 'Физико-технический')
        ) AS "Grou"
FROM Student
WHERE ID_St IN
    (SELECT Id_St
    FROM Exzamen
    WHERE Ball LIKE 5 AND Predmet LIKE 'Иностр. язык')

6.	Подсчитать сколько различных предметов сдавалось в сессию.
SELECT COUNT(Predmet) AS 'Количество' 
FROM Exzamen
WHERE Ball NOT LIKE ""

7.	Напишите запрос, который выводит средний балл по экзаменам Васильевой.
SELECT (SELECT FIO
    FROM Student
    WHERE FIO LIKE 'Васильева Ольга Степановна') AS 'ФИО',
    AVG(Ball) AS 'Ср. балл'
FROM Exzamen    

8.	Определите сколько человек учится на каждой специальности.
SELECT Special, COUNT(FIO) AS 'Кол-во учеников'
FROM Student 
GROUP BY Special

10.	Вывести список фамилий студентов получивших 5 баллов по дифференциальным уравнениям.
SELECT FIO
FROM Student
WHERE ID_St IS
    (SELECT Id_St
    FROM Exzamen
    WHERE Predmet LIKE 'Дифференциальные уравнения' AND Ball LIKE 5
    )

11.	Напишите запрос, который покажет,  сколько экзаменов  сдавал Шутов.

    )

12.	Выведите название кафедры студентов, не сдавших хотя бы один экзамен.
SELECT Kafedra
FROM G
WHERE "Group" IN
    (SELECT "Group"
    FROM Student
    WHERE ID_St IN
        (SELECT Id_St
        FROM Exzamen
        WHERE Ball IS NULL
        )
    )

13.	Подсчитать количество кафедр на каждом факультете.
SELECT DISTINCT Decanat, COUNT(Kafedra) AS 'Кол-во'
FROM Kafedra
GROUP BY Decanat
